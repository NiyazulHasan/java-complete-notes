Q) Write a java program to display Number of digits, sum of digits and it is even or odd?

input:
	NEX123
output:
	Number of Digits : 3
	Sum of Digits : 6
	It is even number 


class Test  
{
	public static void main(String[] args) 
	{
		String str="NEX123";
		
		String newStr = str.replaceAll("[^0-9]","");
		
		System.out.println("No of digits :"+newStr.length());
		
		int sum=0;
		for(int i=0;i<newStr.length();i++)
		{
			int n = Character.getNumericValue(newStr.charAt(i));
			sum+=n;
		}
		System.out.println("Sum of digits :"+sum);
	System.out.println((sum%2==0)?"It is even number":"It is odd number");
	}
}

Q) Write a java program to check given email is valid or invalid?

input:
	raja@gmail.com 
output:
	valid

{
	public static void main(String[] args) 
	{
		String str="raja@gmail.com";
		
		int atIndex = str.indexOf('@');
		int dotIndex = str.indexOf('.');
		
		if(atIndex!=-1 && dotIndex!=-1 && atIndex<dotIndex)

		else
			System.out.println("Invalid");
	}

Q) Write a java program to find out maximum integer value from given string?

input:
	1kg mango is 100 rupees and 2kg of apple is 200 rupees

output:
	200 

class Test  
{
	public static void main(String[] args) 
	{
		String str="1kg mango is 100 rupees and 2kg of apple is 200 rupees";
		
		String[] sarr = str.split("\\D+");
		
		int max = Integer.MIN_VALUE;  //-2147483648 
		
		for(String s : sarr)
		{
			int n = Integer.parseInt(s);
			if(n>max)
			{
				max=n;
			}
		}
		System.out.println(max);
	}
}

Q) Write a java program to sort the string?

	ebcad
output:
	abcde

class Test  
{
	public static void main(String[] args) 
	{
		String str="ebcad";

		System.out.println(str); 
	}

Q) Write a java program to convert lowercase to uppercase and uppercase to lowercase?

input:
	IHubTaLEnt
output:
	ihUBtAleNT


class Test  
{
	public static void main(String[] args) 
	{
		String str="IHubTaLEnt";
		
		char[] carr = str.toCharArray();
		
		for(int i=0;i<carr.length;i++)
		{
			if(carr[i]>='A' && carr[i]<='Z')
			{
				carr[i] = (char)(carr[i]+32);
			}
			else if(carr[i]>='a' && carr[i]<='z')
			{
				carr[i] = (char)(carr[i]-32);
			}
		}
		str = new String(carr);
		System.out.println(str);
	}
}
